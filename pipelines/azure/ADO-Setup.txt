# Variable groups
export ADO_VARGROUP_NONPROD=MyProject_NONPROD
export ADO_VARGROUP_PROD=MyProject_PROD

# ----------------------------------------------------
# Set up your Azure DevOps CLI environment and Project
# ----------------------------------------------------

# Connect to your Azure DevOps instance
az login                                                        # Opens a browser-based interactive authentication session

# Create an Azure DevOps project to host your DataStage assets and Azure build pipelines
az devops project create \
    --name $ADO_PROJECT \
    --description 'MettleCI-enabled DataStage project' \
    --org $ADO_ORGANIZATION_URL \
    --source-control git \
    --visibility private

# Set the default organisation and project for the Azure CLI environment
az devops configure \
    --defaults \
    organization=$ADO_ORGANIZATION_URL \
    project=$ADO_PROJECT


# ----------------------
# Agents and Agent Pools
# ----------------------

# Agent pools can be created easily using the Azure DevOps UI.  Neither the number of pools you create,
# nor the names you give them, are relevant to your MettleCI-enabled pipelines as jobs are automatically
# assigned to agents by Azure DevOps based on the demands required by each of your pipelines' steps and
# the matching capabilities advertised by your agents.

# The definition of Agents requires you to install one or more self-hosted Azure agents on a suitably equipped
# host (see https://learn.microsoft.com/en-us/azure/devops/pipelines/agents/windows-agent?view=azure-devops)
# and associate the agent(s) with a relevant agent pool. 

# List your agent pools
az pipelines pool list \
    --organization $ADO_ORGANIZATION_URL \
    --output table

# Select (or create) your preferred agent pool and assign its ID to $ADO_AGENT_POOL_id
export ADO_AGENT_POOL_id=$(az pipelines pool list \
    --organization $ADO_ORGANIZATION_URL \
    --query '[?name == `MyFavouriteAgentPool`].id' \
    --output tsv)

# List your agent pools' constituent agents
az pipelines agent list \
    --organization $ADO_ORGANIZATION_URL \
    --pool-id $ADO_AGENT_POOL_id \
    --output table


# ------------
# Environments
# ------------

# The creation of Deployment Environments is not currently supported by the Azure CLI.
# Environments are created by the supplied pipelines as they are references. i.e., if you try and
# run a MettleCI deployment to an environment called MyQualityAssurance then environment of that 
# name will be automatically created.  Once a deploy environment has been created (either manually
# using the Azure UI or automatically by reference from a pipeline) you can then configure its 
# 'Approvals and checks' to restrict deployment to that environment as required.


# ------------
# Repositories
# ------------

# Create DataStage repository
export ADO_REPO_DATASTAGE_id=`az repos create \
  --name $ADO_REPOSITORY_DATASTAGE \
  --org $ADO_ORGANIZATION_URL \
  --project $ADO_PROJECT \
  --query 'id' `

# Import DataStage repository from source ($ADO_GIT_SOURCE_URL_DATASTAGE)
az repos import create \
  --git-source-url $ADO_GIT_SOURCE_URL_DATASTAGE \
  --org $ADO_ORGANIZATION_URL \
  --project $ADO_PROJECT \
  --repository $ADO_REPOSITORY_DATASTAGE \
  --requires-authorization`

# Create Compliance repository
export ADO_REPO_COMPLIANCE_id=`az repos create \
  --name $ADO_REPOSITORY_COMPLIANCE \
  --org $ADO_ORGANIZATION_URL \
  --project $ADO_PROJECT \
  --query 'id' `

# Create Compliance repository & import from source ($ADO_GIT_SOURCE_URL_COMPLIANCE)
az repos import create \
  --git-source-url $ADO_GIT_SOURCE_URL_COMPLIANCE \
  --org $ADO_ORGANIZATION_URL \
  --project $ADO_PROJECT \
  --repository $ADO_REPOSITORY_COMPLIANCE \
  --requires-authorization`


# ---------------
# Variable Groups
# ---------------

# Non-production

# Create Non-Production Variable Group
export ADO_VARGROUP_NONPROD_id=`az pipelines variable-group create \
  --name $ADO_VARGROUP_NONPROD \
  --variable \
        AgentMettleCmd=C:\MettleCI\cli\mettleci.cmd \
        AgentMettleHome=C:\MettleCI \
        EngineUnitTestBaseDir=/opt/dm/mci \
        IISDomainName=MyNonProductionServicesTier.MyOrganization.com:59445 \
        IISEngineName=MyNonProductionEngineTier.MyOrganization.com \
        IISProjectTemplateDir=/opt/IBM/InformationServer/Server/Template \
        IISUsername=isadmin \
        IISVersion=11.7 \
        MCIUsername=mciworkb \
  --authorize true \
  --description 'Non-production pipeline variables' \
  --organization $ADO_ORGANIZATION_URL \
  --project $ADO_PROJECT  \
  --query 'id' `

# IISPassword secret variable
az pipelines variable-group variable create \
    --group-id $ADO_VARGROUP_NONPROD_id \
    --org $ADO_ORGANIZATION_URL \
    --project $ADO_PROJECT \
    --secret true \
    --name IISPassword \
    --value MyPassword

# MCIPassword secret variable
az pipelines variable-group variable create \
    --group-id $ADO_VARGROUP_NONPROD_id \
    --org $ADO_ORGANIZATION_URL \
    --project $ADO_PROJECT \
    --secret true \
    --name MCIPassword \
    --value MyPassword

# Production

# ---------
# Pipelines
# ---------

# Create DevOps CI pipeline
az pipelines create \
    --org $ADO_ORGANIZATION_URL \
    --project $ADO_PROJECT \
    --name MyDataStage-DevOps-CI \
    --description 'DevOps CI' \
    --repository $ADO_REPOSITORY_DATASTAGE \
    --branch master \
    --yml-path /devops-ci.yml \
    --repository-type tfsgit

# Create HotFix CI pipeline
az pipelines create \
    --org $ADO_ORGANIZATION_URL \
    --project $ADO_PROJECT \
    --name MyDataStage-HotFix-CI \
    --description 'HotFix CI' \
    --repository $ADO_REPOSITORY_DATASTAGE \
    --branch master \
    --yml-path /hotfix-ci.yml \
    --repository-type tfsgit

# Create a HotFix Deploy pipeline
az pipelines create \
    --org $ADO_ORGANIZATION_URL \
    --project $ADO_PROJECT \
    --name MyDataStage-HotFix-Deploy \
    --description 'HotFix Deploy' \
    --repository $ADO_REPOSITORY_DATASTAGE \
    --branch master \
    --yml-path /hotfix-deploy.yml \
    --repository-type tfsgit
